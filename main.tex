% Set the document class to "article" with a 12pt font size
\documentclass[12pt]{article}

% Use various packages for geometry, font encoding, text positioning, colors, and additional functionalities
\usepackage[margin=1in,top=1.5in,bottom=1in,headheight=1.75in]{geometry} % Set margins
\usepackage[T1]{fontenc} % Use T1 font encoding
\usepackage[absolute,overlay]{textpos} % For absolute text positioning
\usepackage{xcolor} % For color definitions
\usepackage{fancyhdr, graphicx, enumitem, parskip, mathptmx}% For headers/footers, graphics, lists, and correct paragraph spacing
\usepackage{datetime} % For date formatting

\setlength{\parskip}{1em}

% Define a new date format
\newdateformat{monthyeardate}{%
  \twodigit{\THEDAY}~\monthname[\THEMONTH]~\THEYEAR}

% Define a custom color
\definecolor{headercolor}{HTML}{121292}

% Set up the page style
\pagestyle{fancy}
\fancyfoot{} % Remove the default page number from the footer

% Remove the horizontal line in the header
\renewcommand{\headrulewidth}{0pt}

% Define a new command for colored text with a specific font style
\newcommand{\letterhead}[1]{
    \textcolor{headercolor}{\small\textsc{#1}}\rm
}

% Define a new environment for memos with a specific text block size and position
\newenvironment{memoblock}{
    \begin{textblock*}{\textwidth}(1in,1.75in)
    \noindent
}{
    \end{textblock*}
}
% Define a new environment for custom enumerated lists
\newenvironment{memolist}[1][]{
    \begin{enumerate}[labelsep=0.4em,leftmargin=0pt,itemindent=14pt,#1]
}{
    \end{enumerate}
}

% Define a nested list environment with specific formatting
\newenvironment{submemolist}{
    \begin{enumerate}[labelsep=0.2em,leftmargin=1.2em,itemindent=14pt,label=\alph*.~ ]
}{
    \end{enumerate}
}

% Define an environment for signatures with specific positioning and text block size
\newenvironment{signatureblock}{
    \TPoptions{absolute=false}
    \begin{flushleft}
    \begin{textblock*}{3.5in}(3.75in,4em)
    \raggedright
}{
    \end{textblock*}
    \end{flushleft}
}

% Define different headers/footers for the first and subsequent pages
\fancypagestyle{firstpage}{
    \fancyhf{}
    \fancyhead[L]{\begin{textblock*}{1in}(.5in,.5in)\includegraphics[width=1in]{"DAF Seal.png"}\end{textblock*}}
    \fancyhead[C]{
        \letterhead{\textbf{DEPARTMENT OF THE AIR FORCE}}\\
        \letterhead{123d EXAMPLE MEMORANDUM SQUADRON (ACC)}\\
        \letterhead{FORT GEORGE G. MEADE, MARYLAND}
    }
    \renewcommand{\headrulewidth}{0pt}
}

\fancypagestyle{default}{
    \fancyhf{}
    \renewcommand{\headrulewidth}{0pt}
}

\begin{document}
% Apply the first page style
\thispagestyle{firstpage}

% Begin the memo environment
\begin{memoblock}

% Align the date to the right, with a fixed space from the top edge
\hfill\monthyeardate\today

% Memorandum Purpose/Recipient, Source, and Subject
\noindent MEMORANDUM FOR~~RECORD

\noindent FROM:~~123 EMS/ABC

\noindent SUBJECT:~~Use of the Official Memorandum Template

% Begin the custom enumerated list environment
\begin{memolist}
    \item \textbf{Purpose}: This memorandum template has been developed to standardize the format and ensure compliance with the Department of the Air Force (DAF) Tongue \& Quill guidelines. The use of this template will ensure consistency and professionalism in all official communications.
    
    \item \textbf{Scope}: This template is intended for use by all personnel within the 315th Cyberspace Operations Squadron for drafting official memorandums. It provides a clear structure and layout that is easy to follow and ensures all necessary elements are included.
    
    \item \textbf{Instructions for Use}:
        \begin{submemolist}
            \item \textbf{Header and Footer}: The header includes the official seal and unit information. The footer has been left blank to accommodate or CUI/FOUO statements if required.
            \item \textbf{Date Format}: The date should be aligned to the right and formatted using the month, day, and year.
            \item \textbf{Content Structure}: Each memorandum should include a title, from line, subject line, and the body text. Use numbered points for clarity and organization.
        \end{submemolist}
    
    \item \textbf{Benefits}: Utilizing this standardized template will:
        \begin{submemolist}
            \item Enhance the professional appearance of documents.
            \item Ensure compliance with DAF standards.
            \item Improve readability and organization of information.
        \end{submemolist}
    
    \item \textbf{Conclusion}: Adopting this template across the squadron will contribute to more efficient and effective communication. All personnel are encouraged to familiarize themselves with this format and use it for all future memorandum documents.
\end{memolist}

% Begin the signature environment
\begin{signatureblock}
    Evelyn Aurora Lee, TSgt, USAF\\
    Duty Title
\end{signatureblock}

% End the memo environment
\end{memoblock}

\end{document}
