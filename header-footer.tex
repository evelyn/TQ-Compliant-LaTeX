% -------- # Layout Adjustment

% Custom layout adjustments to dynamically modify text height and top margin
\newlength\newtop
\setlength{\newtop}{0.75in} % Set a custom top margin offset (0.75 inches)

% The \@outputpage command is patched to adjust text height and top margin dynamically.
% The \global commands update the global values for text height and top margin.
% The \dimexpr command recalculates the dimensions of the text height and top margin:
%   - It increases text height by the value of \newtop
%   - It decreases top margin by the value of \newtop
% This ensures that the layout adjusts automatically without manually resetting the margin or text height every time.
\makeatletter
\patchcmd\@outputpage{\global \@colht \textheight}{%
    \global\textheight=\dimexpr\textheight+\newtop\relax%
    \global\topmargin=\dimexpr\topmargin-\newtop\relax%
    \global\@colht\textheight%
    \global\newtop\z@}{}{\err} % Reset \newtop after adjustment
\makeatother
