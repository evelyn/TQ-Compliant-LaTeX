% Define the extra space (two non-breaking spaces) as a length
\newcommand{\spaces}{~~}
\newlength{\extraSpaceLength}

\newcommand{\setcolumnwidth}[2]{
    \newlength{#1}%
    \settowidth{#1}{#2}%
    \addtolength{#1}{\extraSpaceLength} % Add extra space to the column width
}
